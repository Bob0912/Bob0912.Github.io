源代码：

#include<stdio.h>#include<math.h>
int main(){ int x1,x2,x3,y1,y2,y3; float dis; scanf(“%d %d %d %d %d %d”,&x1,&y1,&x2,&y2,&x3,&y3); dis = sqrt(pow((x2 - x1) * 1.0, 2)+pow((y2 - y1) * 1.0, 2)); dis = dis+sqrt(pow((x3 - x1) * 1.0, 2)+pow((y3 - y1) * 1.0, 2)); dis = dis+sqrt(pow((x2 - x3) * 1.0, 2)+pow((y2 - y3) * 1.0, 2)); printf(“%.2f”,dis); return 0;}
